    双哈希

    选两组不同的basebasebase和ppp，只有当两组的哈希值的相同是，才判断为相同。这样可以有效降低冲突的概率（平方级别）。

    int a1[N], a2[N];//记录字符串的哈希值
int c1[N], c2[N];//记录base的i次方
void hashstr(char s[])
{
    int len = strlen(s);//字符串长度
    int res = 0;
    c1[0] = c2[0] = 1;
    for (int i = 1; i <= len; i++)
    {
        res = (res * base1 + (s[i] - 'a' + 1)) % p1;
        c1[i] = c1[i - 1] * base1 % p1;
        a1[i] = res;
    }
    res = 0;
    for (int i = 1; i <= len; i++)
    {
        res = (res * base2 + (s[i] - 'a' + 1)) % p2;
        c2[i] = c2[i - 1] * base2 % p2;
        a2[i] = res;
    }
}
pii calsubstr(int l, int r)//询问子串的哈希值
{
    int hashval1 = (a1[r] - 1ll * a1[l - 1] * c1[r - l + 1] % p1 + p1) % p1;
    int hashval2 = (a2[r] - 1ll * a2[l - 1] * c2[r - l + 1] % p2 + p2) % p2;
    return {hashval1, hashval2};
}

typedef long long ll;
typedef pair<int, int> pii;
typedef pair<ll, ll> pll;
typedef unsigned long long u64;
const int N = 1e5 + 100, M = 7e5;
const int mod = 1e9 + 7;
const double pi = acos(-1);
ll n, m;
struct node
{
    int f, v, w;
} a[N];
int p[2 * N];
bool cmp(struct node a, struct node b)
{
    return a.w > b.w;
}
int find(int x)
{
    if (x != p[x])
        p[x] = find(p[x]);
    return p[x];
}
void solve()
{
    int ans;
    cin >> n >> m;
    for (int i = 1; i <= m; i++)
        cin >> a[i].f >> a[i].v >> a[i].w;
    sort(a + 1, a + 1 + m, cmp);
    for (int i = 1; i <= 2 * n; i++)
        p[i] = i;
    for (int i = 1; i <= m; i++)
    {
        auto u = a[i].f, v = a[i].v;
        if (find(u) == find(v) || find(u + n) == find(v + n))
        {
            ans = a[i].w;
            break;
        }
        p[find(u)] = find(v + n), p[find(u + n)] = find(v);
    }
    cout << ans << endl;
}
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int _ = 1;
    // cin >> _;
    while (_--)
        solve();
    return 0;
}

    1.维护可重复贡献问题

    Balanced Lineup维护区间最值

    #include <bits/stdc++.h>
#define endl '\n'
#define x first
#define y second
#define stop cout << "?" << '\n'
using namespace std;
typedef long long ll;
typedef pair<int, int> pii;
typedef pair<ll, ll> pll;
typedef unsigned long long u64;
const int N = 1e5 + 100, M = 7e5;
const int mod = 1e9 + 7;
const double pi = acos(-1);
ll n, q;
int a[N];
struct node
{
    int mi, mx;
} seg[4 * N];
void update(int id)
{
    seg[id].mi = min(seg[id * 2].mi, seg[id * 2 + 1].mi);
    seg[id].mx = max(seg[id * 2].mx, seg[id * 2 + 1].mx);
}
void build(int id, int l, int r)
{
    if (l == r)
        seg[id].mi = seg[id].mx = a[l];
    else
    {
        int mid = l + r >> 1;
        build(id * 2, l, mid);
        build(id * 2 + 1, mid + 1, r);
        update(id);
    }
}
pii query(int id, int l, int r, int ql, int qr)
{
    if (l == ql && r == qr)
        return {seg[id].mi, seg[id].mx};
    int mid = l + r >> 1;
    if (qr <= mid)
        return query(id * 2, l, mid, ql, qr);
    else if (ql > mid)
        return query(id * 2 + 1, mid + 1, r, ql, qr);
    else
    {
        auto t1 = query(id * 2, l, mid, ql, mid), t2 = query(id * 2 + 1, mid + 1, r, mid + 1, qr); // 特别要注意这个地方!
        return {min(t1.x, t2.x), max(t1.y, t2.y)};
    }
}
void solve()
{
    cin >> n >> q;
    for (int i = 1; i <= n; i++)
        cin >> a[i];
    build(1, 1, n);
    while (q--)
    {
        int l, r;
        cin >> l >> r;
        // cout << l << " " << r << endl;
        auto ans = query(1, 1, n, l, r);
        cout << ans.y - ans.x << endl;
    }
}
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int _ = 1;
    // cin >> _;
    while (_--)
        solve();
    return 0;
}

    2.单点修改

    梦迹

    维护区间和,单点修改(下标不能为1)

    #include <bits/stdc++.h>
#define endl '\n'
#define x first
#define y second
#define stop cout << "?" << '\n'
using namespace std;
typedef long long ll;
typedef pair<int, int> pii;
typedef pair<ll, ll> pll;
typedef unsigned long long u64;
const int N = 4e5 + 100, M = 7e5;
const int mod = 1e9 + 7;
const double pi = acos(-1);
ll n, q, w;
ll a[N];
ll cnt[N];
struct info
{
    ll val;
} seg[4 * N];
void update(int id)
{
    seg[id].val = seg[id * 2].val + seg[id * 2 + 1].val;
}
void build(int id, int l, int r)
{
    if (l == r)
        seg[id].val = 0;
    else
    {
        int mid = l + r >> 1;
        build(id * 2, l, mid);
        build(id * 2 + 1, mid + 1, r);
        update(id);
    }
}
// 节点为id,对应区间为[l,r],a[pos]->val
void change(int id, int l, int r, int pos, int val)
{
    if (l == r) // 叶子节点
        seg[id].val = val;
    else
    {
        int mid = l + r >> 1;
        if (pos <= mid)
            change(id * 2, l, mid, pos, val);
        else
            change(id * 2 + 1, mid + 1, r, pos, val);
        update(id);
    }
}
// 查询的区间[ql,qr]
ll query(int id, int l, int r, int ql, int qr)
{
    if (l == ql && r == qr)
        return seg[id].val;
    int mid = l + r >> 1;
    if (qr <= mid)
        return query(id * 2, l, mid, ql, qr);
    else if (ql > mid)
        return query(id * 2 + 1, mid + 1, r, ql, qr);
    else
        return query(id * 2, l, mid, ql, mid) + query(id * 2 + 1, mid + 1, r, mid + 1, qr);
}
void solve()
{
    ll ans = 0;
    cin >> n >> q >> w;
    build(1, 1, 3e5 + 100);
    for (int i = 1; i <= n; i++)
    {
        cin >> a[i], cnt[a[i]]++;
        change(1, 1, 3e5 + 100, a[i] + 1, cnt[a[i]]);
        ans += query(1, 1, 3e5 + 100, 1, w - a[i] + 1);
    }
    while (q--)
    {
        ll p, x;
        cin >> p >> x;
        cnt[a[p]]--, cnt[x]++;
        ans -= query(1, 1, 3e5 + 100, 1, w - a[p] + 1);
        change(1, 1, 3e5 + 100, a[p] + 1, cnt[a[p]]);
        change(1, 1, 3e5 + 100, x + 1, cnt[x]);
        ans += query(1, 1, 3e5 + 100, 1, w - x + 1);
        a[p] = x;
        cout << ans << endl;
    }
}
int main()
{
    ios::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
    int _ = 1;
    // cin >> _;
    while (_--)
        solve();
    return 0;
}
